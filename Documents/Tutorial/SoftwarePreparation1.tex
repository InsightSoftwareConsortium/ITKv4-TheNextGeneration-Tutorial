\begin{frame}
\frametitle{Virtual Machines Preparation}
\begin{itemize}
\item Get DVD / USB Memory Stick
\item Install VirtualBox from it
\item Import the VirtualMachine file
\item Boot the Virtual Machine
\item Log in
\item Get familiar with directories
\end{itemize}
\end{frame}
